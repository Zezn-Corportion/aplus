نتحدث اليوم عن كيفية ربط ملف جافاسكريبت او css بصفحة الويب


الدالة ربط_ملف_جافا_سكريبت

تاخذ رابط ملف جافا سكريبت وتقوم بربطه مع صفحة الويب

من_مكتبة الويب استدعاء * 
هنا = __هنا__

انشاء_دالة_باسم الصفحة_الرئيسية() :
    ربط_ملف_جافا_سكريبت(الرابط = هنا + "min.js")
    اضافة_نص("مرحبا")

الصفحات = [الصفحة_الرئيسية]

تشغيل_السيرفر(الصفحات = الصفحات , المنفذ = 1000)

سيتم ربط ملف الجافا سكريبت علما ان هنا يقصد فيه المكان الذي يوجد فيه كود الموقع


الدالة تشغيل_جافا_سكريبت

هذه الدالة تاخذ كود جافا سكريبت وتقوم بتشغيله مباشرة

تشغيل_جافا_سكريبت("console.log('hi')")

الدالة ربط_ملف_سي_اس_اس

تستخدم هذه الدالة لربط ملف سي اس اس بالصفحة

من_مكتبة الويب استدعاء * 
هنا = __هنا__

انشاء_دالة_باسم الصفحة_الرئيسية() :
    ربط_ملف_سي_اس_اس(الرابط =  "http://example.com/min.css")
    اضافة_نص("مرحبا")

الصفحات = [الصفحة_الرئيسية]

تشغيل_السيرفر(الصفحات = الصفحات , المنفذ = 1000)
